66 //Single producer, single consumer pattern implementation.
67 template <class T, class TAlloc = AkRingBufferAllocatorDefault>
68 class AkRingBuffer
69 {
70 public:
71  AkRingBuffer()
72  : m_nbItems(0)
73  , m_readIndex(0)
74  , m_writeIndex(0)
75  , m_nbReadableItems(0)
76  {
77  }
78 
79  AKRESULT Init(AkUInt32 nbItems)
80  {
81  m_nbItems = nbItems;
82  m_readIndex = 0;
83  m_writeIndex = 0;
84  m_nbReadableItems = 0;
85 
86  m_data = reinterpret_cast<T*>(TAlloc::Alloc(m_nbItems * sizeof(T)));
87  if (m_data == NULL)
88  {
89  return AK_InsufficientMemory;
90  }
91  return AK_Success;
92  }
93 
94  void Term()
95  {
96  m_nbItems = 0;
97  m_readIndex = 0;
98  m_writeIndex = 0;
99  m_nbReadableItems = 0;
100 
101  if (m_data != NULL)
102  {
103  TAlloc::Free(reinterpret_cast<void*>(m_data));
104  m_data = NULL;
105  }
106  }
107 
108  // Reset ringbuffer to initial state without freeing memory. Not thread-safe.
109  void Reset()
110  {
111  m_readIndex = 0;
112  m_writeIndex = 0;
113  m_nbReadableItems = 0;
114  }
115 
116  // ---- Producer ---- //
117 
118  AkUInt32 GetWriteIndex() const
119  {
120  return m_writeIndex;
121  }
122 
123  T* GetWritePtr()
124  {
125  return &m_data[m_writeIndex];
126  }
127 
128  void IncrementWriteIndex(AkUInt32 nbItems)
129  {
130  AKASSERT(GetNbWritableItems() >= nbItems);
131 
132  m_writeIndex = (m_writeIndex + nbItems) % m_nbItems;
133 
134  AkAtomicAdd32(&m_nbReadableItems, nbItems);
135  }
136 
137  void WriteDataToRing(T* in_pSrc, AkUInt32 in_nbItems)
138  {
139  AKASSERT(GetNbWritableItems() >= in_nbItems);
140  AKASSERT(in_nbItems <= m_nbItems);
141 
142  // no wrapping
143  if (m_writeIndex + in_nbItems <= m_nbItems)
144  {
145  AKPLATFORM::AkMemCpy(m_data + m_writeIndex, in_pSrc, in_nbItems * sizeof(T));
146  }
147  else // wrapping past end (split memcpy)
148  {
149  AkUInt32 uItemsBeforeWrap = m_nbItems - m_writeIndex;
150  AkUInt32 uItemsAfterWrap = in_nbItems - uItemsBeforeWrap;
151  AKPLATFORM::AkMemCpy(m_data + m_writeIndex, in_pSrc, uItemsBeforeWrap * sizeof(T));
152  AKPLATFORM::AkMemCpy(m_data, in_pSrc + uItemsBeforeWrap, uItemsAfterWrap * sizeof(T));
153 
154  }
155  m_writeIndex = (m_writeIndex + in_nbItems);
156  m_writeIndex -= m_writeIndex >= m_nbItems ? m_nbItems : 0;
157 
158  AkAtomicAdd32(&m_nbReadableItems, in_nbItems);
159  }
160 
161  // ---- Consumer ----
162 
163  AkUInt32 GetReadIndex() const
164  {
165  return m_readIndex;
166  }
167 
168  const T* GetReadPtr() const
169  {
170  return &m_data[m_readIndex];
171  }
172 
173  // Peek at any item between the read and write pointer without advancing the read pointer.
174  const T* Peek(AkUInt32 uOffset) const
175  {
176  AKASSERT(GetNbReadableItems() > uOffset);
177  AkUInt32 uReadIndex = (m_readIndex + uOffset) % m_nbItems;
178  return &m_data[uReadIndex];
179  }
180 
181  void IncrementReadIndex(AkUInt32 nbItems)
182  {
183  AKASSERT(GetNbReadableItems() >= nbItems);
184 
185  m_readIndex = (m_readIndex + nbItems) % m_nbItems;
186 
187  AkAtomicSub32(&m_nbReadableItems, nbItems);
188  }
189 
190  AkUInt32 GetNbReadableItems() const
191  {
192  return (AkUInt32)AkAtomicLoad32(const_cast<AkAtomic32*>(&m_nbReadableItems));
193  }
194 
195  AkUInt32 GetNbWritableItems() const
196  {
197  return m_nbItems - GetNbReadableItems();
198  }
199 
200  AkUInt32 Size() const
201  {
202  return m_nbItems;
203  }
204 
205  void ReadDataFromRing(T* in_pSrc, AkUInt32 in_nbItems)
206  {
207  AKASSERT(GetNbReadableItems() >= in_nbItems);
208  AKASSERT(in_nbItems <= m_nbItems);
209 
210  // no wrapping
211  if (m_readIndex + in_nbItems <= m_nbItems)
212  {
213  AKPLATFORM::AkMemCpy(in_pSrc, m_data + m_readIndex, in_nbItems * sizeof(T));
214  }
215  else // wrapping past end (split memcpy)
216  {
217  AkUInt32 uItemsBeforeWrap = m_nbItems - m_readIndex;
218  AkUInt32 uItemsAfterWrap = in_nbItems - uItemsBeforeWrap;
219  AKPLATFORM::AkMemCpy(in_pSrc, m_data + m_readIndex, uItemsBeforeWrap * sizeof(T));
220  AKPLATFORM::AkMemCpy(in_pSrc + uItemsBeforeWrap, m_data, uItemsAfterWrap * sizeof(T));
221  }
222  m_readIndex = (m_readIndex + in_nbItems);
223  m_readIndex -= m_readIndex >= m_nbItems ? m_nbItems : 0;
224 
225  AkAtomicSub32(&m_nbReadableItems, in_nbItems);
226  }
227 
228  // Warning: requires external locking to prevent concurrent Grow+Read in a multi-threaded scenario.
229  // Like the rest of the class, assumes a single writing thread.
230  bool Grow(AkUInt32 in_uGrowBy)
231  {
232  AkUInt32 uTargetItems = m_nbItems + in_uGrowBy;
233  if (T* pNewData = reinterpret_cast<T*>(TAlloc::Alloc(uTargetItems * sizeof(T))))
234  {
235  AkUInt32 uReadableItems = GetNbReadableItems();
236  if (uReadableItems)
237  {
238  if (m_readIndex >= m_writeIndex)
239  {
240  // insert new free space in the middle of the buffer.
241 
242  if (m_writeIndex)
243  AKPLATFORM::AkMemCpy(pNewData, m_data, sizeof(T) * m_writeIndex);
244 
245  AKPLATFORM::AkMemCpy(pNewData + m_readIndex + in_uGrowBy, m_data + m_readIndex, sizeof(T) * (m_nbItems - m_readIndex));
246  m_readIndex += in_uGrowBy;
247  }
248  else
249  {
250  // insert new free space at the end of the buffer.
251 
252  AKPLATFORM::AkMemCpy(pNewData + m_readIndex, m_data + m_readIndex, sizeof(T) * uReadableItems);
253  }
254  }
255 
256  TAlloc::Free(reinterpret_cast<void*>(m_data));
257  m_data = pNewData;
258  m_nbItems = uTargetItems;
259 
260  return true;
261  }
262  else
263  {
264  return false;
265  }
266  }
267 
268 private:
269  T* m_data{ nullptr };
270 
271  AkUInt32 m_nbItems;
272  AkUInt32 m_readIndex;
273  AkUInt32 m_writeIndex;
274  AkAtomic32 m_nbReadableItems;
275 };
