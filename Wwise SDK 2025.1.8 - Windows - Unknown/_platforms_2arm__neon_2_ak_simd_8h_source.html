263 static AkForceInline AKSIMD_V4F32 AKSIMD_CONVERT_V4F16_TO_V4F32_HELPER(uint16x4_t vecs16)
264 {
265 #if defined(AK_CPU_ARM_64) && (defined(__GNUC__) && !defined(__llvm__)) && (__GNUC__ < 6 || __GNUC__ == 6 && __GNUC_MINOR__ < 1)
266 
267  // float16 intrinsics were added in gcc 6.1, and we still use gcc4.9 on Android
268  // all compilers that we support for arm64 - i.e. clang/msvc - support the intrinsics just fine
269  float32x4_t ret;
270  __asm__("fcvtl %0.4s, %1.4h" \
271  : "=w"(ret) \
272  : "w"(vecs16)
273  );
274  return ret;
275 #elif defined(AK_CPU_ARM_64) && defined(AK_MAC_OS_X)
276  float16x4_t vecf16 = vreinterpret_f16_s16(vecs16);
277  float32x4_t ret = vcvt_f32_f16(vecf16);
278  uint32x4_t signData = vshll_n_u16(vand_u16(vecs16, vdup_n_u16(0x8000)), 16);
279  ret = vorrq_u32(vreinterpretq_s32_f32(ret), signData);
280  return ret;
281 #elif defined(AK_CPU_ARM_64)
282  return vcvt_f32_f16(vreinterpret_f16_s16(vecs16));
283 #else
284  uint32x4_t vecExtended = vshlq_n_u32(vmovl_u16(vecs16), 16);
285  uint32x4_t expMantData = vandq_u32(vecExtended, vdupq_n_u32(0x7fff0000));
286  uint32x4_t expMantShifted = vshrq_n_u32(expMantData, 3); // shift so that the float16 exp/mant is now split along float32's bounds
287 
288  // Determine if value is denorm or not, and use that to determine how much exponent should be scaled by, and if we need post-scale fp adjustment
289  uint32x4_t isDenormMask = vcltq_u32(expMantData, vdupq_n_u32(0x03ff0000));
290  uint32x4_t exponentIncrement = vbslq_u32(isDenormMask, vdupq_n_u32(0x38800000), vdupq_n_u32(0x38000000));
291  uint32x4_t postIncrementAdjust = vandq_u32(isDenormMask, vdupq_n_u32(0x38800000));
292 
293  // Apply the exponent increment and adjust
294  uint32x4_t expMantScaled = vaddq_u32(expMantShifted, exponentIncrement);
295  uint32x4_t expMantAdj = vreinterpretq_u32_f32(vsubq_f32(vreinterpretq_f32_u32(expMantScaled), vreinterpretq_f32_u32(postIncrementAdjust)));
296 
297  // if fp16 val was inf or nan, preserve the inf/nan exponent field (we can just 'or' inf-nan
298  uint32x4_t isInfnanMask = vcgtq_u32(expMantData, vdupq_n_u32(0x7bffffff));
299  uint32x4_t infnanExp = vandq_u32(isInfnanMask, vdupq_n_u32(0x7f800000));
300  uint32x4_t expMantWithInfNan = vorrq_u32(expMantAdj, infnanExp);
301 
302  // reincorporate the sign
303  uint32x4_t signData = vandq_u32(vecExtended, vdupq_n_u32(0x80000000));
304  float32x4_t assembledFloat = vreinterpretq_f32_u32(vorrq_u32(signData, expMantWithInfNan));
305  return assembledFloat;
306 #endif
307 }

310 /// Converts the 4 full-precision floats vector to 4 half-precision floats
311 /// occupying the lower bits and leaving the upper bits as zero
312 static AkForceInline AKSIMD_V4I32 AKSIMD_CONVERT_V4F32_TO_V4F16(AKSIMD_V4F32 vec)
313 {
314 #if defined(AK_CPU_ARM_64) && (defined(__GNUC__) && !defined(__llvm__)) && (__GNUC__ < 6 || __GNUC__ == 6 && __GNUC_MINOR__ < 1)
315  // float16 intrinsics were added in gcc 6.1, and we still use gcc4.9 on Android
316  // all compilers that we support for arm64 - i.e. clang/msvc - support the intrinsics just fine
317  int32x4_t ret;
318  __asm__("fcvtn %1.4h, %1.4s\n" \
319  "\tmov %0.8b, %1.8b" \
320  : "=w"(ret) \
321  : "w"(vec)
322  );
323  return ret;
324 #elif defined(AK_CPU_ARM_64) && defined(AK_MAC_OS_X)
325  float16x4_t ret = vcvt_f16_f32(vec);
326  uint16x4_t signData = vshrn_n_u32(vandq_u32(vreinterpretq_u32_f32(vec), vdupq_n_u32(0x80000000)), 16);
327  ret = vorr_u16(vreinterpret_s16_f16(ret), signData);
328  return vreinterpretq_s32_s16(vcombine_s16(vreinterpret_s16_f16(ret), vdup_n_s16(0)));
329 #elif defined(AK_CPU_ARM_64)
330  return vreinterpretq_s32_s16(vcombine_s16(vreinterpret_s16_f16(vcvt_f16_f32(vec)), vdup_n_s16(0)));
331 #else
332  uint32x4_t signData = vandq_u32(vreinterpretq_u32_f32(vec), vdupq_n_u32(0x80000000));
333  uint32x4_t unsignedVec = vandq_u32(vreinterpretq_u32_f32(vec), vdupq_n_u32(0x7fffffff));
334 
335  // do the processing for values that will be denormed in float16
336  // Add 0.5 to get value within range, and rounde; then move mantissa data up
337  float32x4_t denormedVec = vaddq_f32(vreinterpretq_f32_u32(unsignedVec), vdupq_n_f32(0.5f));
338  uint32x4_t denormResult = vshlq_n_u32(vreinterpretq_u32_f32(denormedVec), 16);
339 
340  // processing for values that will be normal in float16
341  uint32x4_t subnormMagic = vdupq_n_u32(0xC8000FFF); // -131072 + rounding bias
342  uint32x4_t normRoundPart1 = vaddq_u32(unsignedVec, subnormMagic);
343  uint32x4_t mantLsb = vshlq_n_u32(unsignedVec, 31 - 13);
344  uint32x4_t mantSignExtendLsb = vshrq_n_u32(mantLsb, 31); // Extend Lsb so that it's -1 when set
345  uint32x4_t normRoundPart2 = vsubq_u32(normRoundPart1, mantSignExtendLsb); // and subtract the sign-extended bit to finish rounding up
346  uint32x4_t normResult = vshlq_n_u32(normRoundPart2, 3);
347 
348  // Combine the norm and subnorm paths together
349  uint32x4_t normalMinimum = vdupq_n_u32((127 - 14) << 23); // smallest float32 that yields a normalized float16
350  uint32x4_t denormMask = vcgtq_u32(normalMinimum, unsignedVec);
351 
352  uint32x4_t nonNanFloat = vbslq_u32(denormMask, denormResult, normResult);
353 
354  // apply inf/nan check
355  uint32x4_t isNotInfNanMask = vcltq_u32(unsignedVec, vdupq_n_u32(0x47800000)); // test if exponent bits are zero or not
356  uint32x4_t mantissaData = vandq_u32(unsignedVec, vdupq_n_u32(0x007fffff));
357  uint32x4_t isNanMask = vmvnq_u32(vceqq_f32(vec, vec)); // mark the parts of the vector where we have a mantissa (i.e. NAN) as 0xffffffff
358  uint32x4_t nantissaBit = vandq_u32(isNanMask, vdupq_n_u32(0x02000000)); // set the NaN mantissa bit if mantissa suggests this is NaN
359  uint32x4_t infData = vandq_u32(vmvnq_u32(mantissaData), vdupq_n_u32(0x7c000000)); // grab the exponent data from unsigned vec with no mantissa
360  uint32x4_t infNanData = vorrq_u32(infData, nantissaBit); // if we have a non-zero mantissa, add the NaN mantissa bit
361 
362  uint32x4_t resultWithInfNan = vbslq_u32(isNotInfNanMask, nonNanFloat, infNanData); // and combine the results
363 
364  // reincorporate the original sign
365  uint32x4_t signedResult = vorrq_u32(signData, resultWithInfNan);
366 
367  // store results packed in lower 64 bits, and set upper 64 to zero
368  uint16x8x2_t resultZip = vuzpq_u16(vreinterpretq_u16_u32(signedResult), vdupq_n_u16(0));
369  return vreinterpretq_s32_u16(resultZip.val[1]);
370 #endif
371 }

704 AkForceInline AKSIMD_V4F32 AKSIMD_SQRT_V4F32(AKSIMD_V4F32 x)
