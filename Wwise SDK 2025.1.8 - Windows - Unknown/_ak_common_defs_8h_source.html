168 // Build a 32 bit class identifier based on the Plug-in type,
169 // CompanyID and PluginID.
170 //
171 // Parameters:
172 // - in_pluginType: A value from enum AkPluginType (4 bits)
173 // - in_companyID: CompanyID as defined in the Plug-in's XML file (12 bits)
174 // * 0-63: Reserved for Audiokinetic
175 // * 64-255: Reserved for clients' in-house Plug-ins
176 // * 256-4095: Assigned by Audiokinetic to third-party plug-in developers
177 // - in_pluginID: PluginID as defined in the Plug-in's XML file (16 bits)
178 // * 0-32767: Set freely by the Plug-in developer
179 #define AKMAKECLASSID( in_pluginType, in_companyID, in_pluginID ) \
180  ( (in_pluginType) + ( (in_companyID) << 4 ) + ( (in_pluginID) << ( 4 + 12 ) ) )
181

183 #define AKGETCOMPANYIDFROMCLASSID( in_classID ) ( ( (in_classID) & 0x0000FFF0 ) >> 4 )
184 #define AKGETPLUGINIDFROMCLASSID( in_classID ) ( ( (in_classID) & 0xFFFF0000 ) >> ( 4 + 12 ) )

396  /// Convert a channel, identified by a single channel bit, to a buffer index used in GetChannel() below, for a given channel config.
397  /// Standard indexing follows channel bit order (see AkSpeakerConfig.h). Pipeline/buffer indexing is the same but the LFE is moved to the end.
398  static inline AkUInt32 StandardToPipelineIndex(
399  AkChannelConfig in_channelConfig, ///< Channel configuration.
400  AkUInt32 in_uChannelIdx ///< Channel index in standard ordering to be converted to pipeline ordering.
401  )
402  {
403  if ( in_channelConfig.HasLFE() )
404  {
405  AKASSERT( in_channelConfig.eConfigType == AK_ChannelConfigType_Standard ); // in_channelConfig.HasLFE() would not have returned true otherwise.
406  AKASSERT( AK::GetNumNonZeroBits( in_channelConfig.uChannelMask ) );
407  AkUInt32 uIdxLFE = AK::GetNumNonZeroBits( ( AK_SPEAKER_LOW_FREQUENCY - 1 ) & in_channelConfig.uChannelMask );
408  if ( in_uChannelIdx == uIdxLFE )
409  return in_channelConfig.uNumChannels - 1;
410  else if ( in_uChannelIdx > uIdxLFE )
411  return in_uChannelIdx - 1;
412  }
413 
414  return in_uChannelIdx;
415  }

417  /// Get the buffer of the ith channel. To be used with de-interleaved buffers only.
418  ///
419  /// Access to channel data is most optimal through this method. Use whenever the
420  /// speaker configuration is known, or when an operation must be made independently
421  /// for each channel.
422  /// \remarks When using a standard Wwise pipeline configuration, use ChannelBitToIndex() to convert channel bits to buffer indices.
423  /// \return Address of the buffer of the ith channel.

447  /// Can be used to transform an incomplete into a complete buffer with valid data.
448  /// The invalid frames are made valid (zeroed out) for all channels and the validFrames count will be made equal to uMaxFrames.
449  void ZeroPadToMaxFrames()
450  {
451  AKASSERT(pData != nullptr || MaxFrames() == 0);
452  // The following members MUST be copied locally due to multi-core calls to this function.
453  const AkUInt32 uNumChannels = NumChannels();
454  const AkUInt32 uNumCurrentFrames = AkMin(uValidFrames, MaxFrames());
455  const AkUInt32 uNumZeroFrames = MaxFrames() - uNumCurrentFrames;
456  if ( uNumZeroFrames )
457  {
458  AKASSERT(pData != nullptr);
459  for ( AkUInt32 i = 0; i < uNumChannels; ++i )
460  {
461  memset( GetChannel(i) + uNumCurrentFrames, 0, uNumZeroFrames * sizeof(AkSampleType) );
462  }
463  uValidFrames = MaxFrames();
464  }
465  }
