311  /// Output a debug message on the console (variadic function).
312  /// Use MaxSize to specify the size to reserve for the message.
313  /// Warning: On Win32, OutputDebugMsgV with wchar_t will truncate the string
314  /// to MaxSize.
315  template <int MaxSize = 256>
316  inline void OutputDebugMsgV(const wchar_t* in_pszFmt, ...)
317  {
318  // No equivalent of snprintf for wide string so truncate if necessary
319  // vswprintf returns the number of characters written and not the numbers
320  // that would be written
321 
322  wchar_t* msg = (wchar_t*)AkAlloca(MaxSize * sizeof(wchar_t));
323  msg[MaxSize - 1] = '\0';
324 
325  va_list args;
326  va_start(args, in_pszFmt);
327  vswprintf(msg, MaxSize, in_pszFmt, args);
328  va_end(args);
329 
330  OutputDebugMsg(msg);
331  }
332 
333  /// Output a debug message on the console (variadic function).
334  /// Use MaxSize to specify the expected size of the message.
335  /// If the required size is superior to MaxSize, a new string
336  /// will be allocated on the stack with the required size.
337  template <int MaxSize = 256>
338  inline void OutputDebugMsgV(const char* in_pszFmt, ...)
339  {
340  int size = 0;
341  {
342  // Try with a reasonable guess first
343  char msg[MaxSize];
344  msg[MaxSize - 1] = '\0';
345 
346  va_list args;
347  va_start(args, in_pszFmt);
348  size = vsnprintf(msg, MaxSize, in_pszFmt, args);
349  va_end(args);
350 
351  // If it was enough, print to debug log
352  if (0 <= size && size <= MaxSize)
353  {
354  OutputDebugMsg(msg);
355  return;
356  }
357  }
358 
359  // Else, we need more memory to prevent truncation
360  {
361  // size + 1 more char for the last '\0'
362  size++;
363 
364  char* msg = (char*)AkAlloca((size) * sizeof(char));
365  msg[size - 1] = '\0';
366 
367  va_list args;
368  va_start(args, in_pszFmt);
369  vsnprintf(msg, size, in_pszFmt, args);
370  va_end(args);
371 
372  OutputDebugMsg(msg);
373  }
374  }
