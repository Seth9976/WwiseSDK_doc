81  // AkBitScanForward returns how many 0s there are until the least-significant-bit is set
82  // (or the length of the param if the value is zero)
83  // e.g. 0b0000`0000`0001`0000 would return 4
84 #if defined(__clang__) || defined(__GNUC__)
85  AkForceInline AkUInt32 AkBitScanForward64(AkUInt64 in_bits)
86  {
87  return in_bits ? __builtin_ctzll(in_bits) : 64;
88  }
89 #elif defined _MSC_VER && (defined AK_CPU_X86_64 || defined AK_CPU_ARM_64)
90  AkForceInline AkUInt32 AkBitScanForward64(AkUInt64 in_bits)
91  {
92  unsigned long ret = 0;
93  _BitScanForward64(&ret, in_bits);
94  return in_bits ? ret : 64;
95  }
96 #else
97  AkForceInline AkUInt32 AkBitScanForward64(AkUInt64 in_bits)
98  {
99  if (in_bits == 0) return 64;
100  AkUInt32 ret = 0;
101  if ((in_bits & 0x00000000FFFFFFFFULL) == 0) { ret += 32; in_bits >>= 32; }
102  if ((in_bits & 0x000000000000FFFFULL) == 0) { ret += 16; in_bits >>= 16; }
103  if ((in_bits & 0x00000000000000FFULL) == 0) { ret += 8; in_bits >>= 8; }

206  /// Computes the next power of two given a value.
207  /// \return next power of two.
208  AkForceInline AkUInt32 GetNextPowerOfTwo(AkUInt32 in_uValue)
209  {
210  in_uValue--;
211  in_uValue |= in_uValue >> 1;
212  in_uValue |= in_uValue >> 2;
213  in_uValue |= in_uValue >> 4;
214  in_uValue |= in_uValue >> 8;
215  in_uValue |= in_uValue >> 16;
216  in_uValue++;
217  return in_uValue;
218  }

230  // given an input x (with only lower 8 bits set) translate it into 8 byte-sized bools
231  // (PDEP but for bitfield->bool conversion)
232  AkForceInline AkUInt64 AkBitfieldToBytes64(AkUInt64 x)
233  {
234  AkUInt64 depBits = ((x * 0x8040201008040201) & 0x8080808080808080) >> 7;
235 #if defined(__clang__) || defined (__GNUC__)
236  return __builtin_bswap64(depBits);
237 #elif defined _MSC_VER
238  return _byteswap_uint64(depBits);
239 #else
240  x = (x & 0x00000000FFFFFFFF) << 32 | (x & 0xFFFFFFFF00000000) >> 32;
241  x = (x & 0x0000FFFF0000FFFF) << 16 | (x & 0xFFFF0000FFFF0000) >> 16;
242  x = (x & 0x00FF00FF00FF00FF) << 8 | (x & 0xFF00FF00FF00FF00) >> 8;
243  return x;
244 #endif
245  }
