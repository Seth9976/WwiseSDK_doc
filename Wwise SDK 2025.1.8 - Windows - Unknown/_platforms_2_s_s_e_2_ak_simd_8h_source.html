249 /// Faked in-place vector horizontal add - each element will represent sum of all elements
250 /// \akwarning
251 /// Don't expect this to be very efficient.
252 /// \endakwarning
253 static AkForceInline AKSIMD_V4F32 AKSIMD_HORIZONTALADD_V4F32(AKSIMD_V4F32 vVec)
254 {
255  __m128 vAb = _mm_shuffle_ps(vVec, vVec, 0xB1);

268 /// Cross-platform SIMD multiplication of 2 complex data elements with interleaved real and imaginary parts
269 static AkForceInline AKSIMD_V4F32 AKSIMD_COMPLEXMUL_V4F32( const AKSIMD_V4F32 vCIn1, const AKSIMD_V4F32 vCIn2 )
270 {
271  const AKSIMD_V4F32 vSign = { -0.f, 0.f, -0.f, 0.f };
272 
273  AKSIMD_V4F32 vTmp1 = AKSIMD_SHUFFLE_V4F32( vCIn1, vCIn1, AKSIMD_SHUFFLE(2,2,0,0));
274  vTmp1 = AKSIMD_MUL_V4F32( vTmp1, vCIn2 );
275  AKSIMD_V4F32 vTmp2 = AKSIMD_SHUFFLE_V4F32( vCIn1, vCIn1, AKSIMD_SHUFFLE(3,3,1,1));
276  vTmp2 = AKSIMD_XOR_V4F32( vTmp2, vSign );
277  vTmp2 = AKSIMD_MADD_V4F32( vTmp2, AKSIMD_SHUFFLE_BADC( vCIn2 ), vTmp1 );
278  return vTmp2;
279 }

378 // Given four pointers, gathers 32-bits of data from each location,
379 // deinterleaves them as 16-bits of each, and sign-extends to 32-bits
380 // e.g. (*addr[0]) := (b a)
381 // e.g. (*addr[1]) := (d c)
382 // e.g. (*addr[2]) := (f e)
383 // e.g. (*addr[3]) := (h g)
384 // return struct has
385 // val[0] := (g e c a)
386 // val[1] := (h f d b)
387 static AkForceInline AKSIMD_V4I32X2 AKSIMD_GATHER_V4I32_AND_DEINTERLEAVE_V4I32X2(AkInt16* addr3, AkInt16* addr2, AkInt16* addr1, AkInt16* addr0)
388 {
389  __m128i data[4] = {
390  _mm_set1_epi32(*(AkInt32*)addr0),
391  _mm_set1_epi32(*(AkInt32*)addr1),
392  _mm_set1_epi32(*(AkInt32*)addr2),
393  _mm_set1_epi32(*(AkInt32*)addr3),
394  };
395 
396  __m128i group[2] = {
397  _mm_unpacklo_epi32(data[0], data[1]),
398  _mm_unpacklo_epi32(data[2], data[3]),
399  };
400 
401  __m128i shuffle = _mm_unpacklo_epi64(group[0], group[1]);
402 
403  AKSIMD_V4I32X2 ret{
404  _mm_srai_epi32(_mm_slli_epi32(shuffle, 16), 16),
405  _mm_srai_epi32(shuffle, 16)
406  };
407  return ret;
408 }

578 static AkForceInline AKSIMD_V4F32 AKSIMD_CONVERT_V4F16_TO_V4F32_HELPER(AKSIMD_V4I32 vec)
579 {
580  __m128i expMantData = _mm_and_si128(vec, _mm_set1_epi32(0x7fff0000));
581  __m128i expMantShifted = _mm_srli_epi32(expMantData, 3); // shift so that the float16 exp/mant is now split along float32's bounds
582 
583  // magic number to get scale fp16 exp range into fp32 exp range (also renormalize any denorms)
584  __m128i expMantFloat = _mm_castps_si128(_mm_mul_ps(_mm_castsi128_ps(expMantShifted), _mm_castsi128_ps(_mm_set1_epi32(0x77800000))));
585 
586  // if fp16 val was inf or nan, preserve the inf/nan exponent field (we can just 'or' the new inf-bits into the attempt at scaling from inf previously)
587  __m128i infnanCheck = _mm_cmpgt_epi32(expMantData, _mm_set1_epi32(0x7bffffff));
588  __m128i infnanExp = _mm_and_si128(infnanCheck, _mm_set1_epi32(255 << 23));
589  __m128i expMantWithInfNan = _mm_or_si128(expMantFloat, infnanExp);
590 
591  // reincorporate the sign
592  __m128i signData = _mm_and_si128(vec, _mm_set1_epi32(0x80000000));
593  __m128 assembledFloat = _mm_castsi128_ps(_mm_or_si128(signData, expMantWithInfNan));
594  return assembledFloat;

597 /// Converts the 4 full-precision floats vector to 4 half-precision floats
598 /// occupying the lower bits and leaving the upper bits as zero
599 static AkForceInline AKSIMD_V4I32 AKSIMD_CONVERT_V4F32_TO_V4F16(AKSIMD_V4F32 vec)
600 {
601  __m128i signData = _mm_and_si128(_mm_castps_si128(vec), _mm_set1_epi32(0x80000000));
602  __m128i unsignedVec = _mm_and_si128(_mm_castps_si128(vec), _mm_set1_epi32(0x7fffffff));
603 
604  // do the processing for values that will be denormed in float16
605  // Add 0.5 to get value within range, and rounde; then move mantissa data up
606  __m128 denormedVec = _mm_add_ps(_mm_castsi128_ps(unsignedVec), _mm_set1_ps(0.5f));
607  __m128i denormResult = _mm_slli_epi32(_mm_castps_si128(denormedVec), 16);
608 
609  // processing for values that will be normal in float16
610  __m128i subnormMagic = _mm_set1_epi32(0xC8000FFF); // -131072 + rounding bias
611  __m128i normRoundPart1 = _mm_add_epi32(unsignedVec, subnormMagic);
612  __m128i mantLsb = _mm_slli_epi32(unsignedVec, 31 - 13);
613  __m128i mantSignExtendLsb = _mm_srai_epi32(mantLsb, 31); // Extend Lsb so that it's -1 when set
614  __m128i normRoundPart2 = _mm_sub_epi32(normRoundPart1, mantSignExtendLsb); // and subtract the sign-extended bit to finish rounding up
615  __m128i normResult = _mm_slli_epi32(normRoundPart2, 3);
616 
617  // Combine the norm and subnorm paths together
618  __m128i normalMinimum = _mm_set1_epi32((127 - 14) << 23); // smallest float32 that yields a normalized float16
619  __m128i denormMask = _mm_cmpgt_epi32(normalMinimum, unsignedVec);
620 
621  __m128i nonNanFloat = _mm_or_si128(_mm_and_si128(denormMask, denormResult), _mm_andnot_si128(denormMask, normResult));
622 
623  // apply inf/nan check
624  __m128i isNotInfNanMask = _mm_cmplt_epi32(unsignedVec, _mm_set1_epi32(0x47800000)); // test if the value will be greater than the max representable by float16
625  __m128i mantissaData = _mm_and_si128(unsignedVec, _mm_set1_epi32(0x007fffff));
626  __m128i isNanMask = _mm_cmpgt_epi32(unsignedVec, _mm_set1_epi32(0x7F800000)); // mark the parts of the vector where we have a mantissa (i.e. NAN) as 0xffffffff
627  __m128i nantissaBit = _mm_and_si128(isNanMask, _mm_set1_epi32(0x02000000)); // set the NaN mantissa bit if mantissa suggests this is NaN
628  __m128i infData = _mm_andnot_si128(mantissaData, _mm_set1_epi32(0x7c000000)); // grab the exponent data from unsigned vec with no mantissa
629  __m128i infNanFloat = _mm_or_si128(infData, nantissaBit); // if we have a non-zero mantissa, add the NaN mantissa bit
630 
631  __m128i resultWithInfNan = _mm_or_si128(_mm_and_si128(isNotInfNanMask, nonNanFloat), _mm_andnot_si128(isNotInfNanMask, infNanFloat));
632 
633  // reincorporate the original sign
634  __m128i signedResult = _mm_or_si128(signData, resultWithInfNan);
635 
636  // store results packed in lower 64 bits, and set upper 64 to zero
637  __m128i resultEpi16Lo = _mm_shufflelo_epi16(signedResult, 0xD); // move 16b ints (x,x,x,x,d,c,b,a) down to (x,x,x,x,x,x,d,b)
638  __m128i resultEpi16Hi = _mm_shufflehi_epi16(signedResult, 0xD); // move 16b ints (h,g,f,e,x,x,x,x) down to (x,x,h,f,x,x,x,x)
639  __m128 resultEpi16 = _mm_shuffle_ps(_mm_castsi128_ps(resultEpi16Lo), _mm_castsi128_ps(resultEpi16Hi), 0xE4); // combine - (x, x, h, f, x, x, d, b)
640  __m128i result = _mm_castps_si128(_mm_shuffle_ps(resultEpi16, _mm_setzero_ps(), 0x8)); // reshuffle with zero - (0,0,0,0,h,f,d,b)
641 
642  return result;
643 }
